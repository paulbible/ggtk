41  inline double calculateSimilarity(const boost::unordered_set<std::string> &termsA, const boost::unordered_set<std::string> &termsB){
42  // Get the induced set of terms for each set
43  boost::unordered_set<std::string> inducedTermSetA = getExtendedTermSet(termsA);
44  boost::unordered_set<std::string> inducedTermSetB = getExtendedTermSet(termsB);
45  // Calculate union and intersection
46  boost::unordered_set<std::string> union_set = SetUtilities::set_union(inducedTermSetA, inducedTermSetB);
47  boost::unordered_set<std::string> intersection_set = SetUtilities::set_intersection(inducedTermSetA, inducedTermSetB);
48 
49  double union_sum = 0.0;
50  double intersection_sum = 0.0;
51  boost::unordered_set<std::string>::iterator it, end;
52  // calculate sum for union set
53  it = union_set.begin();
54  end = union_set.end();
55  for (; it != end; ++it){
56  union_sum += _icMap[*it];
57  }
58  // Calculate sum for intersection set
59  it = intersection_set.begin();
60  end = intersection_set.end();
61  for (; it != end; ++it){
62  intersection_sum += _icMap[*it];
63  }
64 
65  //if the union is 0, return 0. No division by 0.
66  if (union_sum == 0.0){
67  return 0.0;
68  }
69  else{
70  return intersection_sum / union_sum;
71  }
72  }

93  inline boost::unordered_set<std::string> getExtendedTermSet(const boost::unordered_set<std::string> &terms){
94  boost::unordered_set<std::string> inducedSet;
95  boost::unordered_set<std::string>::iterator it, end;
96  it = terms.begin();
97  end = terms.end();
98  for (; it != end; ++it){
99  std::string term = *it;
100  // add the new terms to the set using union and the ancestors from the go graph
101  inducedSet = SetUtilities::set_union(inducedSet, _graph->getAncestorTerms(term));
102  inducedSet.insert(term);
103  }
104  return inducedSet;
