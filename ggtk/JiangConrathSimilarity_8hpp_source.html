50  inline double calculateTermSimilarity(std::string goTermA, std::string goTermB){
51  //if the terms do not exit return 0.0 similarity
52  if(!_icMap.hasTerm(goTermA) || !_icMap.hasTerm(goTermB)){
53  return 0.0;
54  }
55 
56  //if not from same ontology, return 0;
57  if(_goGraph->getTermOntology(goTermA) != _goGraph->getTermOntology(goTermB)){
58  return 0;
59  }
60 
61  //create 2 sets
62  boost::unordered_set<std::string> ancestorsA = _goGraph->getAncestorTerms(goTermA);
63  ancestorsA.insert(goTermA);
64  boost::unordered_set<std::string> ancestorsB = _goGraph->getAncestorTerms(goTermB);
65  ancestorsB.insert(goTermB);
66 
67  //if either set is empty, return 0
68  if(ancestorsA.size() == 0 || ancestorsB.size() == 0){
69  return 0.0;
70  }
71 
72  double maxIC;
73 
74  //select the correct ontology normalization factor
75  GO::Onto ontoType = _goGraph->getTermOntology(goTermA);
76  if(ontoType == GO::BP){
77  maxIC = -std::log(_icMap.getMinBP());
78  }else if(ontoType == GO::MF){
79  maxIC = -std::log(_icMap.getMinMF());
80  }else{
81  maxIC = -std::log(_icMap.getMinCC());
82  }
83 
84  //get the MICA
85  std::string mica = getMICA(ancestorsA,ancestorsB);
86 
87  double dist = (_icMap[goTermA] + _icMap[goTermB] - 2*_icMap[mica]);
88 
89  return 1 - (dist/(2.0*maxIC));
90  }

110  inline std::string getMICA(boost::unordered_set<std::string> &ancestorsA,boost::unordered_set<std::string> &ancestorsB){
111  //get the first term as a start
112  std::string mica = *ancestorsA.begin();
113  double max = -1.0;
114 
115  //always traverse the shortest list
116  if(ancestorsA.size() > ancestorsB.size()){
117  ancestorsA.swap(ancestorsB);
118  }
119 
120  //loop over shorter list
121  boost::unordered_set<std::string>::iterator iter;
122  for(iter = ancestorsA.begin(); iter != ancestorsA.end(); ++iter){
123  //current string
124  std::string currentTerm = *iter;
125 
126  if(ancestorsB.find(currentTerm) == ancestorsB.end()){
127  //continue, not a shared ancestor term
128  continue;
129  }
130 
131  //if new max, update
132  if(_icMap[currentTerm] > max){
133  mica = currentTerm;
134  max = _icMap[currentTerm];
135  }
136  }
137 
138  return mica;
139  }
