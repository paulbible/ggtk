60  inline double oneSidedRawPvalue_hyper(size_t sample, size_t success,size_t population,size_t test_value){
61  double sum = 0.0;
62  boost::math::hypergeometric dist(sample,success,population);
63  for(size_t i = test_value; i <= sample && i <= success; ++i){
64  double prob = boost::math::pdf(dist,i);
65  sum += prob;
66  }
67  return sum;

77  inline double enrichmentSignificance(GoGraph *go, AnnotationData *data,
78  boost::unordered_set<std::string> &genes,
79  const std::string &term)
80  {
81  boost::unordered_set<std::string> termGenes = getDescendantGenes(go, data, term);

83

85  return 1.0;

88  size_t sampleSize = genes.size();
89  size_t sampleWithTerm = sharedGenes.size();
