43  inline double calculateSimilarity(const boost::unordered_set<std::string> &termsA, const boost::unordered_set<std::string> &termsB){
44  // Get the induced set of terms for each set
45  boost::unordered_set<std::string> inducedTermSetA = getExtendedTermSet(termsA);
46  boost::unordered_set<std::string> inducedTermSetB = getExtendedTermSet(termsB);
47  // Calculate union and intersection
48  boost::unordered_set<std::string> intersection_set = SetUtilities::set_intersection(inducedTermSetA, inducedTermSetB);
49 
50  double intersection_sum = calcICSum(intersection_set);
51  double setA_sum = calcICSum(inducedTermSetA);
52  double setB_sum = calcICSum(inducedTermSetB);
53 
54 
55  //if the union is 0, return 0. No division by 0.
56  if (setA_sum + setB_sum == 0.0){
57  return 0.0;
58  }
59  else{
60  if (setA_sum > setB_sum){
61  return intersection_sum / setA_sum;
62  }
63  else{
64  return intersection_sum / setB_sum;
65  }
66 
67  }
68  }

89  inline boost::unordered_set<std::string> getExtendedTermSet(const boost::unordered_set<std::string> &terms){
90  boost::unordered_set<std::string> inducedSet;
91  boost::unordered_set<std::string>::iterator it, end;
92  it = terms.begin();
93  end = terms.end();
94  for (; it != end; ++it){
95  std::string term = *it;
96  // add the new terms to the set using union and the ancestors from the go graph.
97  inducedSet = SetUtilities::set_union(inducedSet, _graph->getAncestorTerms(term));
98  inducedSet.insert(term);
99  }
100  return inducedSet;
101  }

107  inline double calcICSum(const boost::unordered_set<std::string> &terms){
108  boost::unordered_set<std::string>::iterator it, end;
109  it = terms.begin();
110  end = terms.end();
111  double sum = 0;
112  for (; it != end; ++it){
113  sum += _icMap[*it];
