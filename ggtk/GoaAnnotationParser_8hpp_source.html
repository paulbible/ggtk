118  inline bool isFileGood(const std::string &fileName){
119  std::ifstream in(fileName.c_str());
120  if (!in.good()){
121  return false;
122  }

124  //Tokenizer type
125  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
126  //declares a tab separator variable
127  boost::char_separator<char> tab_sep("\t", "", boost::keep_empty_tokens);
128  tokenizer::iterator it;
129 
130  std::size_t count = 0;
131  std::string line;
132 
133  while (in.good() && count < 5){
134  //get next line in 'in' file stream
135  std::getline(in, line);
136 
137  //split line
138  tokenizer tokens(line, tab_sep);
139  //set iterator to first token
140  it = tokens.begin();
141 
142  //always check if empty (must have in linux)
143  if (it == tokens.end()){ continue; }
144 
145  //skip comments if line is not empty and starts with !
146  if (line.at(0) == '!'){ continue; }
147 
148 
149  std::string database, geneName, qualifierStr, goString, evidenceCode, ontology;
150  std::size_t i = 0;
151  for (; it != tokens.end(); ++it){
152  switch (i)
153  {
154  case 0:
155  database = *it;
156  if (database.size() == 0){ return false; }
157  break;
158  case 1:
159  geneName = *it;
160  if (geneName.size() == 0){ return false; }
161  break;
162  case 3:
163  qualifierStr = *it;
164  break;
165  case 4:
166  goString = *it;
167  if (goString.size() == 0){ return false; } // disallow empty go
168  if (goString.substr(0, 3) != "GO:"){ return false; } // disallow bad go term
169  break;
170  case 6:
171  evidenceCode = *it;
172  if (evidenceCode.size() == 0){ return false; }
173  if (GO::evidenceStringToCode(evidenceCode) == GO::ECODE_ERROR){ return false;}
174  break;
175  case 8:
176  ontology = *it;
177  if (ontology.size() == 0){ return false; }
178  break;
179  default:
180  break;
181  }
182  ++i;
183  }
184 
185  ++count;
186  }
187  in.close();
188 
189 
190  if (count < 5){
191  return false;
192  }
193  else{
194  return true;
195  }
