49  inline double calculateTermSimilarity(std::string goTermA, std::string goTermB){
50  //if the terms do not exit return 0.0 similarity
51  if (!_depthMap.hasTerm(goTermA) || !_depthMap.hasTerm(goTermB)){
52  return 0.0;
53  }
54 
55  //if not from same ontology, return 0;
56  if(_goGraph->getTermOntology(goTermA) != _goGraph->getTermOntology(goTermB)){
57  return 0;
58  }
59 
60  //create 2 sets
61  boost::unordered_set<std::string> ancestorsA = _goGraph->getAncestorTerms(goTermA);
62  ancestorsA.insert(goTermA);
63  boost::unordered_set<std::string> ancestorsB = _goGraph->getAncestorTerms(goTermB);
64  ancestorsB.insert(goTermB);
65 
66  //if either set is empty, return 0
67  if(ancestorsA.size() == 0 || ancestorsB.size() == 0){
68  return 0.0;
69  }
70 
71  std::string lca = getLCA(ancestorsA,ancestorsB);
72 
73  //std::cout << "Pekar Staab LCA " << lca << std::endl;
74 
75  double lcaDepth = _depthMap[lca];
76 
77  //std::cout << "Pekar Staab LCA " << lcaDepth << std::endl;
78  //std::cout << "Pekar Staab delta(a,lca) " << (_depthMap[goTermA] - lcaDepth) << std::endl;
79  //std::cout << "Pekar Staab delta(b,lca) " << (_depthMap[goTermB] - lcaDepth) << std::endl;
80 
81 
82  //return the similarity of Pekar Staab
83  return (double)lcaDepth/( (_depthMap[goTermA] - lcaDepth) + (_depthMap[goTermB] - lcaDepth) +lcaDepth);

103  inline std::string getLCA(boost::unordered_set<std::string> &ancestorsA,boost::unordered_set<std::string> &ancestorsB){
104  //get the first term as a start
105  std::string lca = *ancestorsA.begin();
106  //max depth
107  double max = -1.0;
108 
109  //always traverse the shortest list
110  if(ancestorsA.size() > ancestorsB.size()){
111  ancestorsA.swap(ancestorsB);
112  }
113 
114  //loop over shorter list
115  boost::unordered_set<std::string>::iterator iter;
116  for(iter = ancestorsA.begin(); iter != ancestorsA.end(); ++iter){
117  //current string
118  std::string currentTerm = *iter;
119 
120  if(ancestorsB.find(currentTerm) == ancestorsB.end()){
121  //continue, not a shared ancestor term
122  continue;
123  }
124 
125  //if new max, update
126  if(_depthMap[currentTerm] > max){
127  lca = currentTerm;
128  max = _depthMap[currentTerm];
129  }
130  }
131 
132  return lca;
133  }
