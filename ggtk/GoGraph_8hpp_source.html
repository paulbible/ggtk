168  inline void insertTerm(const std::string &termId, const std::string &name, const std::string &description, const std::string &ontology){
169 
170  //term already exists, update its information,
171  if(_nameToIndex.find(termId) != _nameToIndex.end()){
172  std::size_t index = _nameToIndex[termId];
173 
174  //Term needs to be updated
175  //If name is "name", this is a stub, no need to update
176  if(name.compare("name") != 0){
177  _names.at(index) = name;
178  _descriptions.at(index) = description;
179  GoVertex V = boost::vertex(index,_goGraph);
180  _goGraph[V].ontology = GO::ontologyStringToCode(ontology);
181 
182  }
183 
184  }else{
185  //term is new and must be added
186  //map termId to index
187  _nameToIndex[termId] = boost::num_vertices(_goGraph);
188 
189  //add the vertex to the graph, get back a reference to the vertex, V
190  GoVertex V = boost::add_vertex(_goGraph);
191 
192  //set the termId
193  _goGraph[V].termId = termId;
194  //set the ontology
195  _goGraph[V].ontology = GO::ontologyStringToCode(ontology);
196 
197  //add name to list
198  _names.push_back(name);
199  //add description to list
200  _descriptions.push_back(description);
201  }
202  }

759  inline std::size_t getNumComponents(){
760  //Define undirected graph type
761  typedef boost::adjacency_list < boost::vecS, boost::vecS, boost::undirectedS> undirected_graph_t;
762  undirected_graph_t undirected_g;
763 
764  //Make an undirected copy of the graph
765  for (std::size_t i = 0; i < getNumVertices(); ++i){
766  //add graph vertices
767  boost::add_vertex(undirected_g);
768  }
769  boost::graph_traits<Graph>::edge_iterator iter, end;
770  boost::tie(iter, end) = boost::edges(_goGraph);
771  for (; iter != end; ++iter){
772  GoVertex s = boost::source(*iter, _goGraph);
773  GoVertex t = boost::target(*iter, _goGraph);
774  //add edges, undirected
775  boost::add_edge(_vMap[s], _vMap[t], undirected_g);
776  }
777 
778  //calculate the connected components
779  std::vector<std::size_t> componentAssignment(getNumVertices());
780  return boost::connected_components(boost::make_reverse_graph(undirected_g), &componentAssignment[0]);
781  }
