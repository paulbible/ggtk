47  inline double calculateTermSimilarity(std::string goTermA, std::string goTermB){
48  //if the terms do not exit return 0.0 similarity
49  if (!_icMap.hasTerm(goTermA) || !_icMap.hasTerm(goTermB)){
50  return 0.0;
51  }
52  //if not from same ontology, return 0;
53  if(_goGraph->getTermOntology(goTermA) != _goGraph->getTermOntology(goTermB)){
54  return 0;
55  }
56 
57  //create 2 sets
58  boost::unordered_set<std::string> ancestorsA = _goGraph->getAncestorTerms(goTermA);
59  ancestorsA.insert(goTermA);
60  boost::unordered_set<std::string> ancestorsB = _goGraph->getAncestorTerms(goTermB);
61  ancestorsB.insert(goTermB);
62 
63  //if either set is empty, return 0
64  if(ancestorsA.size() == 0 || ancestorsB.size() == 0){
65  return 0.0;
66  }
67 
68  //get the MICA
69  std::string mica = getMICA(ancestorsA,ancestorsB);
70 
71  //return the normalized information content similarity of Lin
72  if(goTermA != goTermB){
73  return 2*_icMap[mica]/(_icMap[goTermA]+_icMap[goTermB]);
74  }else{
75  return 1.0;
76  }
77  }

92  inline std::string getMICA(boost::unordered_set<std::string> &ancestorsA,boost::unordered_set<std::string> &ancestorsB){
93  //get the first term as a start
94  std::string mica = *ancestorsA.begin();
95  double max = -1.0;
96 
97  //always traverse the shortest list
98  if(ancestorsA.size() > ancestorsB.size()){
99  ancestorsA.swap(ancestorsB);
100  }
101 
102  //loop over shorter list
103  boost::unordered_set<std::string>::iterator iter;
104  for(iter = ancestorsA.begin(); iter != ancestorsA.end(); ++iter){
105  //current string
106  std::string currentTerm = *iter;
107 
108  if(ancestorsB.find(currentTerm) == ancestorsB.end()){
109  //continue, not a shared ancestor term
110  continue;
111  }
112 
113  //if new max, update
114  if(_icMap[currentTerm] > max){
115  mica = currentTerm;
116  max = _icMap[currentTerm];
117  }
118  }
119 
120  return mica;
121  }
