136  inline double sharedInformation(const std::string &termA, const std::string &termB){
137  // return 0 for any terms not in the datbase
138  if (!_icMap.hasTerm(termA) || !_icMap.hasTerm(termB)){
139  return 0.0;
140  }
141  // return 0 for terms in different ontologies
142  if (_goGraph->getTermOntology(termA) != _goGraph->getTermOntology(termB)){
143  return 0.0;
144  }
145 
146  Accumulators::MeanAccumulator meanIC;

151  for(;iter != cda.end(); ++iter){
152  //std::cout << _icMap[*iter] << std::endl;
153  meanIC(_icMap[*iter]);
154  }
155 
156  return Accumulators::extractMean(meanIC);
