136  inline
137  NCList(std::vector<GenomicRegion> &regions){
138 
139  _items = std::vector<NCNode>();
140  //reserve genes.size(), good guess to the number of nodes needed.
141  _items.reserve(regions.size());
142  _totalFeatures = regions.size();
143 
144  //sort by the start position of the gene, biased by gene size.
145  std::sort(regions.begin(),regions.end());
146 
147  std::vector<GenomicRegion>::iterator rItr;
148 
149  if(regions.size() != 0){
150 
151  //temporary count of genes added to a sub list
152  size_t addCount = 0;
153  //temporary storage for all regions that are contained by another region.
154  std::vector<GenomicRegion> tempList;
155 
156  //current region being examined.
157  GenomicRegion currentRegion;
158 
159  //iterate each region in the list as we create the NCList
160  std::vector<GenomicRegion>::iterator currentPtr = regions.end();
161  for(rItr = regions.begin();rItr != regions.end();++rItr){
162  if(currentPtr == regions.end()){
163  //guarantee currentPtr is the first gene at the beginning
164  currentPtr = rItr;
165  currentRegion = *currentPtr;
166  }
167  else{
168  GenomicRegion rtemp = *rItr;
169  if(currentRegion.contains(rtemp)){
170  //continue adding regions contained by currentRegion to the templist
171  // these will be added later
172  ++addCount;
173  tempList.push_back(rtemp);
174 
175  }else{
176  //the next region is not contained by currentRegion,
177  // add all Regions in the list to an NCNode and
178  // add it to the list
179  NCNode n = NCNode(currentRegion,tempList);
180  _items.push_back(n);
181 
182  currentPtr = rItr;
183  currentRegion = rtemp;
184  tempList.clear();
185  addCount = 0;
186  }
187  }
188  }
189 
190  //Add a Node to the list if it is empty,
191  // if not empty add all regions in the tempList to the last node.
192  if(_items.empty() || !_items.at(_items.size()-1).contains(currentRegion)){
193  _items.push_back(NCNode(currentRegion,tempList));
194  }
195  }
196  }

205  size_t getOverlapIndex(const GenomicRegion r){
206  size_t i = 0;
207  size_t k = _items.size()-1;
208 
209  //use binary search to find the best index for the gene or region
210  while(i < k){
211  size_t j = (i+k)/2;
212  NCNode cNode = _items.at(j);
213 
214  if(cNode.contains(r)){
215  return j;
216  }else{
217  if(r._start < cNode._region._start){
218  k = j;
219  }else{
220  i = j+1;
221  }
222  }
223  }
224  //return the index in the list
225  return i;
226  }
