172  inline std::size_t getNumPaths(const std::string &termA, const std::string &termB){
173  if(_icMap[termA] > _icMap[termB]){
174  return 0;
175  }

180  boost::unordered_map<std::string,std::size_t> pathCountMap;
181 
182  //pathCountMap[termB] = 1;

184  //use dfs to determing the number of paths to the target gene
185  GoGraph::GoVertex startVertex = boost::vertex(0,*subgraph);
186  PathNumDFSVisitor pathNumVis(pathCountMap);
187  boost::depth_first_search(boost::make_reverse_graph(*subgraph),boost::visitor(pathNumVis).root_vertex(startVertex));

206  return 0.0;

216  meanIC(_icMap[*iter]);

283  {
