48  inline void writeSimilarityMatrix(TermSimilarityInterface* termSim, std::string fileName, long ontology_code){
49  GO::Onto ontology = static_cast<GO::Onto>(ontology_code);
50  std::vector<std::string> ontologyTerms = _annoData->getOntologyTerms(_goGraph, ontology);
51 
52  // Initialze a matrix
53  std::size_t nTerms = ontologyTerms.size();
54  std::vector<std::vector<double> > matrix;
55  matrix.reserve(nTerms);
56  for (std::size_t i = 0; i < nTerms; ++i){
57  matrix.push_back(std::vector<double>());
58  for (std::size_t j = 0; j < nTerms; ++j){
59  matrix[i].push_back(0.0);
60  }
61  }
62 
63  // --- Main Calculation ---
64  for (std::size_t i = 0; i < nTerms; ++i){
65  std::string termA = ontologyTerms.at(i);
66  matrix[i][i] = termSim->calculateNormalizedTermSimilarity(termA, termA);
67  for (std::size_t j = i + 1; j < nTerms; ++j){
68  std::string termB = ontologyTerms.at(j);
69  double value = termSim->calculateNormalizedTermSimilarity(termA, termB);
70  //cout << termA << "\t" << termB << "\t" << value << endl;
71  matrix[i][j] = value;
72  matrix[j][i] = value;
73  }
74  }
75 
76  // write the matrix
77  writeMatrix(fileName, matrix, ontologyTerms);
78  }

110  inline void writeMatrix(std::string fname, const std::vector<std::vector<double> > &matrix, const std::vector<std::string> &terms){
111  std::ofstream out(fname.c_str());
112  std::size_t matrix_size = matrix.size();
113  for (std::size_t i = 0; i < matrix_size; ++i){
114  std::string termA = terms.at(i);
115  out << termA;
116  for (std::size_t j = 0; j < matrix_size; ++j){
117  out << "\t" << matrix[i][j];
118  }
119  out << std::endl;
120  }
121  out.close();
122  }
