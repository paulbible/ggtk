264  std::size_t pathCount(const std::string &termA, const std::string &termB){
265  if (_icMap[termA] > _icMap[termB]){
266  return 0;
267  }

284  void visitHelper(const GoGraph::GoVertex &v, GoGraph::Graph* g,
285  boost::unordered_set<std::string> &ancestors,
286  boost::unordered_set<std::string> &finished,
287  boost::unordered_map<std::string, size_t> &pathMap)
288  {
289  size_t childCount = 0;
290  std::string vTerm = (*g)[v].termId;
291  //std::cout << "discover vertex " << vTerm << std::endl;
292 
293  //examine children and recurse
294  GoGraph::InEdgeIterator it, end;
295  for (boost::tie(it, end) = boost::in_edges(v, *g); it != end; ++it){
296  GoGraph::GoVertex child = boost::source(*it, *g);
297  std::string childTerm = (*g)[child].termId;
298  if (!SetUtilities::set_contains(ancestors, childTerm)){
299  continue;
300  }
301  //recurse if child is not finished
302  if (!SetUtilities::set_contains(finished, childTerm)){
303  visitHelper(child, g, ancestors, finished, pathMap);
304  }
305  ++childCount;
306  }
307 
308  //finish vertex
309  finished.insert(vTerm);
310  //std::cout << "finish vertex " << vTerm << ", childred " << childCount << std::endl;
311  if (childCount == 0){
312  pathMap[vTerm] = 1;
313  }
314  else{
315  pathMap[vTerm] = 0;
316  for (boost::tie(it, end) = boost::in_edges(v, *g); it != end; ++it){
317  GoGraph::GoVertex child = boost::source(*it, *g);
318  std::string childTerm = (*g)[child].termId;
319  if (!SetUtilities::set_contains(ancestors, childTerm)){
320  continue;
321  }
322  pathMap[vTerm] += pathMap[childTerm];
323  }
324  }
325  }
