37  inline double calculateSimilarity(const boost::unordered_set<std::string> &termsA, const boost::unordered_set<std::string> &termsB){
38  // Get the induced set of terms for each set
39  boost::unordered_set<std::string> inducedTermSetA = getExtendedTermSet(termsA);
40  boost::unordered_set<std::string> inducedTermSetB = getExtendedTermSet(termsB);
41  // Calculate union and intersection
42  boost::unordered_set<std::string> union_set = SetUtilities::set_union(inducedTermSetA, inducedTermSetB);
43  boost::unordered_set<std::string> intersection_set = SetUtilities::set_intersection(inducedTermSetA, inducedTermSetB);
44 
45  //if the union is 0, return 0. No division by 0.
46  if (union_set.size() == 0){
47  return 0.0;
48  }
49  else{
50  return (double)intersection_set.size() / union_set.size();
51  }
52  }
