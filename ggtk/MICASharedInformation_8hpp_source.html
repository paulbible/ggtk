49  inline double sharedInformation(const std::string &termA,const std::string &termB){
50  // return 0 for any terms not in the datbase
51  if (!_icMap.hasTerm(termA) || !_icMap.hasTerm(termB)){
52  return 0.0;
53  }
54  // return 0 for terms in different ontologies
55  if (_goGraph->getTermOntology(termA) != _goGraph->getTermOntology(termB)){
56  return 0.0;
57  }
58 
59  Accumulators::MaxAccumulator myMax;
60 
61  boost::unordered_set<std::string> ancestorsA = _goGraph->getAncestorTerms(termA);
62  ancestorsA.insert(termA);
63  boost::unordered_set<std::string> ancestorsB = _goGraph->getAncestorTerms(termB);
64  ancestorsB.insert(termB);
65 
66  boost::unordered_set<std::string> sharedAncestors = SetUtilities::set_intersection(ancestorsA,ancestorsB);
67 
68  boost::unordered_set<std::string>::iterator iter = sharedAncestors.begin();
69  for(;iter != sharedAncestors.end(); ++iter){
70  myMax(_icMap[*iter]);
71  }
72 
73  return Accumulators::extractMax(myMax);
74  }
