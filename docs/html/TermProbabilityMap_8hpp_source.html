37  inline TermProbabilityMap(GoGraph* graph, AnnotationData* annoData){
38 
39  //set the default minimum probablity policy for normalization
40  _isSingleAnnoMin = false;
41 
42  //Initialize an annotation list the size of verticies in go, each value is 0
43  _probabilities = std::vector<double>(graph->getNumVertices(),0);
44 
45  //get the (first) root of the ontology.
46  GoGraph::GoVertex root = graph->getRoot();
47 
48  //TESTING
49  //std::cout << root << std::endl;
50  //std::cout << graph->getTermStringIdByIndex(root) << std::endl;
51 
52  //a variable for the cummulative annotaions of the graph
53  std::vector<std::size_t> annotationCounts(graph->getNumVertices(),0);
54 
55  //create the visitor abject
56  dfs_cumulative_annotations_visitor vis(graph ,annoData, &annotationCounts,&_nameToIndex);
57 
58  //get the boost graph from the GoGraph object. Must be done to utilize boost algorithms
59  GoGraph::Graph* g = graph->getGraph();
60 
61  //call the boost depth first search using our custom visitor
62  // revering the graph is necessary otherwise the root vertex would have no edges.
63  boost::depth_first_search(boost::make_reverse_graph(*g), boost::visitor(vis).root_vertex(root));
64 
65  // Traverse the vertices to find the roots for each ontology
66  Accumulators::SimpleAccumulator minMaxBP;
67  Accumulators::SimpleAccumulator minMaxMF;
68  Accumulators::SimpleAccumulator minMaxCC;
69 
70  // Vertex Iterators
71  GoGraph::GoVertexIterator vi,vend;
72  for(boost::tie(vi,vend) = boost::vertices(*g); vi != vend; ++vi){
73  GoGraph::GoVertex v = *vi;
74  std::size_t index = graph->getVertexIndex(v);
75 
76  double counts = (double)annotationCounts.at(index);
77  if(counts == 0.0){
78  continue;
79  }
80 
81  //switch on ontology, find the max for each set
82  switch(graph->getTermOntologyByIndex(index)){
83  case GO::BP:
84  minMaxBP(counts);
85  break;
86 
87  case GO::MF:
88  minMaxMF(counts);
89  break;
90 
91  case GO::CC:
92  minMaxCC(counts);
93  break;
94 
95  case GO::ONTO_ERROR:
96  break;
97  }
98  }//end for, vertex iterator
99 
100  //calculate single annotation minimum normalization factors
101  _bp_normalization_min_1anno = 1.0/Accumulators::extractMax(minMaxBP);
102  _mf_normalization_min_1anno = 1.0/Accumulators::extractMax(minMaxMF);
103  _cc_normalization_min_1anno = 1.0/Accumulators::extractMax(minMaxCC);
104 
105  //calculate minimum annotation minimum normalization factors
106  _bp_normalization_min_minAnno = Accumulators::extractMin(minMaxBP)/Accumulators::extractMax(minMaxBP);
107  _mf_normalization_min_minAnno = Accumulators::extractMin(minMaxMF)/Accumulators::extractMax(minMaxMF);
108  _cc_normalization_min_minAnno = Accumulators::extractMin(minMaxCC)/Accumulators::extractMax(minMaxCC);
109 
110  //Traverse the cummulative annotation vector, convert to probability, add to probability vector
111  for(boost::tie(vi,vend) = boost::vertices(*g); vi != vend; ++vi){
112  GoGraph::GoVertex v = *vi;
113  std::size_t index = graph->getVertexIndex(v);
114  switch (graph->getTermOntologyByIndex(index)){
115  case GO::BP:
116  _probabilities.at(index) = annotationCounts.at(index)/Accumulators::extractMax(minMaxBP);
117  break;
118  case GO::MF:
119  _probabilities.at(index) = annotationCounts.at(index)/Accumulators::extractMax(minMaxMF);
120  break;
121  case GO::CC:
122  _probabilities.at(index) = annotationCounts.at(index)/Accumulators::extractMax(minMaxCC);
123  break;
124  case GO::ONTO_ERROR:
125  break;
126  }
127  }
128  }//end constructor logic
