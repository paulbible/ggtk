167  inline void insertTerm(const std::string &termId, const std::string &name, const std::string &description, const std::string &ontology){
168 
169  //term already exists, update its information,
170  if(_nameToIndex.find(termId) != _nameToIndex.end()){
171  std::size_t index = _nameToIndex[termId];
172 
173  //Term needs to be updated
174  //If name is "name", this is a stub, no need to update
175  if(name.compare("name") != 0){
176  _names.at(index) = name;
177  _descriptions.at(index) = description;
178  GoVertex V = boost::vertex(index,_goGraph);
179  _goGraph[V].ontology = GO::ontologyStringToCode(ontology);
180 
181  }
182 
183  }else{
184  //term is new and must be added
185  //map termId to index
186  _nameToIndex[termId] = boost::num_vertices(_goGraph);
187 
188  //add the vertex to the graph, get back a reference to the vertex, V
189  GoVertex V = boost::add_vertex(_goGraph);
190 
191  //set the termId
192  _goGraph[V].termId = termId;
193  //set the ontology
194  _goGraph[V].ontology = GO::ontologyStringToCode(ontology);
195 
196  //add name to list
197  _names.push_back(name);
198  //add description to list
199  _descriptions.push_back(description);
200  }
201  }

668  inline boost::unordered_set<std::string> getOntologyTerms(GO::Onto ontology){
669  //Use only terms in the annotation database, this will save on space and computation time.
670  std::string rootId;
671  switch (ontology){
672  case GO::BP:
673  rootId = GO::getRootTermBP();
674  break;
675  case GO::MF:
676  rootId = GO::getRootTermMF();
677  break;
678  case GO::CC:

682  default:
683  rootId = "";
684  break;

686  //All Ontology terms are descendants of the root.
687  boost::unordered_set<std::string> ontologyTerms = getDescendantTerms(rootId);
688  // Add the root.
689  ontologyTerms.insert(rootId);
690  return ontologyTerms;
691  }

732  inline std::size_t getNumComponents(){
733  //Define undirected graph type
734  typedef boost::adjacency_list < boost::vecS, boost::vecS, boost::undirectedS> undirected_graph_t;
735  undirected_graph_t undirected_g;
736 
737  //Make an undirected copy of the graph
738  for (std::size_t i = 0; i < getNumVertices(); ++i){
739  //add graph vertices
740  boost::add_vertex(undirected_g);
741  }
742  boost::graph_traits<Graph>::edge_iterator iter, end;
743  boost::tie(iter, end) = boost::edges(_goGraph);
744  for (; iter != end; ++iter){
745  GoVertex s = boost::source(*iter, _goGraph);
746  GoVertex t = boost::target(*iter, _goGraph);
747  //add edges, undirected
748  boost::add_edge(_vMap[s], _vMap[t], undirected_g);
749  }
750 
751  //calculate the connected components
752  std::vector<std::size_t> componentAssignment(getNumVertices());
753  return boost::connected_components(boost::make_reverse_graph(undirected_g), &componentAssignment[0]);
754  }
