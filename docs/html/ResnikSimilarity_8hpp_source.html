106  inline std::string getMICA(boost::unordered_set<std::string> &ancestorsA,boost::unordered_set<std::string> &ancestorsB){
107  //get the first term as a start
108  std::string mica = *ancestorsA.begin();
109  double max = -1.0;
110 
111  //always traverse the shortest list
112  if(ancestorsA.size() > ancestorsB.size()){
113  ancestorsA.swap(ancestorsB);
114  }
115 
116  //loop over shorter list
117  boost::unordered_set<std::string>::iterator iter;
118  for(iter = ancestorsA.begin(); iter != ancestorsA.end(); ++iter){
119  //current string
120  std::string currentTerm = *iter;
121 
122  if(ancestorsB.find(currentTerm) == ancestorsB.end()){
123  //continue, not a shared ancestor term
124  continue;
125  }
126 
127  //if new max, update
128  if(_icMap[currentTerm] > max){
129  mica = currentTerm;
130  max = _icMap[currentTerm];
131  }
132  }
